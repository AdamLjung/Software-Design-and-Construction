329 ismount(path) -> Bool
330
331 Return `true` if `path` is a mount point, `false` otherwise.
332 """
333 function ismount(path...)
334 path = joinpath(path...)
335 isdir(path) || return false
336 s1 = lstat(path)
337 # Symbolic links cannot be mount points
338 islink(s1) && return false
339 parent_path = joinpath(path, "..")
340 s2 = lstat(parent_path)
341 # If a directory and its parent are on different devices, then the
342 # directory must be a mount point
343 (s1.device != s2.device) && return true
344 (s1.inode == s2.inode) && return true
345 false
346 end
