48 function pwd()
49 buf = Base.StringVector(AVG_PATH - 1) # space for null-terminator implied by StringVector
50 sz = RefValue{Csize_t}(length(buf) + 1) # total buffer size including null
51 while true
52 rc = ccall(:uv_cwd, Cint, (Ptr{UInt8}, Ptr{Csize_t}), buf, sz)
53 if rc == 0
54 resize!(buf, sz[])
55 return String(buf)
56 elseif rc == Base.UV_ENOBUFS
57 resize!(buf, sz[] - 1) # space for null-terminator implied by StringVector
58 else
59 uv_error(:cwd, rc)
60 end
61 end

239 rm(path::AbstractString; force::Bool=false, recursive::Bool=false)
240
241 Delete the file, link, or empty directory at the given path. If `force=true` is passed, a
242 non-existing path is not treated as error. If `recursive=true` is passed and the path is a
243 directory, then all contents are removed recursively.
244
245 # Examples
246 ```jldoctest
247 julia> mkpath("my/test/dir");
248
249 julia> rm("my", recursive=true)
250
251 julia> rm("this_file_does_not_exist", force=true)
252
253 julia> rm("this_file_does_not_exist")
254 ERROR: IOError: unlink: no such file or directory (ENOENT)
255 Stacktrace:
256 [...]
257 ```
258 """
259 function rm(path::AbstractString; force::Bool=false, recursive::Bool=false)
260 if islink(path) || !isdir(path)
261 try
262 @static if Sys.iswindows()
263 # is writable on windows actually means "is deletable"
264 if (filemode(lstat(path)) & 0o222) == 0
265 chmod(path, 0o777)
266 end
267 end
268 unlink(path)
269 catch err
270 if force && isa(err, IOError) && err.code==Base.UV_ENOENT
271 return
272 end
273 rethrow()
274 end
275 else
276 if recursive
277 for p in readdir(path)
278 rm(joinpath(path, p), force=force, recursive=true)
279 end
280 end
281 @static if Sys.iswindows()
282 ret = ccall(:_wrmdir, Int32, (Cwstring,), path)
283 else
284 ret = ccall(:rmdir, Int32, (Cstring,), path)
285 end
286 systemerror(:rmdir, ret != 0, extrainfo=path)
287 end

314 function cptree(src::AbstractString, dst::AbstractString; force::Bool=false,
315 follow_symlinks::Bool=false)
316 isdir(src) || throw(ArgumentError("'$src' is not a directory. Use `cp(src, dst)`"))

461 const TEMP_CLEANUP_MIN = Ref(1024)
462 const TEMP_CLEANUP_MAX = Ref(1024)
463 const TEMP_CLEANUP = Dict{String,Bool}()
464 const TEMP_CLEANUP_LOCK = ReentrantLock()
465
466 function temp_cleanup_later(path::AbstractString; asap::Bool=false)
467 lock(TEMP_CLEANUP_LOCK)
468 # each path should only be inserted here once, but if there
469 # is a collision, let !asap win over asap: if any user might
470 # still be using the path, don't delete it until process exit
471 TEMP_CLEANUP[path] = get(TEMP_CLEANUP, path, true) & asap
472 if length(TEMP_CLEANUP) > TEMP_CLEANUP_MAX[]
473 temp_cleanup_purge(false)
474 TEMP_CLEANUP_MAX[] = max(TEMP_CLEANUP_MIN[], 2*length(TEMP_CLEANUP))
475 end
476 unlock(TEMP_CLEANUP_LOCK)
477 return nothing
478 end
479
480 function temp_cleanup_purge(all::Bool=true)
481 need_gc = Sys.iswindows()
482 for (path, asap) in TEMP_CLEANUP
483 try
484 if (all || asap) && ispath(path)
485 need_gc && GC.gc(true)
486 need_gc = false
487 rm(path, recursive=true, force=true)
488 end
489 !ispath(path) && delete!(TEMP_CLEANUP, path)
490 catch ex
491 @warn "temp cleanup" _group=:file exception=(ex, catch_backtrace())
492 end
493 end
494 end

572 tempname(parent=tempdir(); cleanup=true) -> String
573
574 Generate a temporary file path. This function only returns a path; no file is
575 created. The path is likely to be unique, but this cannot be guaranteed due to
576 the very remote posibility of two simultaneous calls to `tempname` generating
577 the same file name. The name is guaranteed to differ from all files already
578 existing at the time of the call to `tempname`.
579
580 When called with no arguments, the temporary name will be an absolute path to a
581 temporary name in the system temporary directory as given by `tempdir()`. If a
582 `parent` directory argument is given, the temporary path will be in that
583 directory instead.
584
585 The `cleanup` option controls whether the process attempts to delete the
586 returned path automatically when the process exits. Note that the `tempname`
587 function does not create any file or directory at the returned location, so
588 there is nothing to cleanup unless you create a file or directory there. If
589 you do and `clean` is `true` it will be deleted upon process termination.
590
591 !!! compat "Julia 1.4"
592 The `parent` and `cleanup` arguments were added in 1.4. Prior to Julia 1.4
593 the path `tempname` would never be cleaned up at process termination.
594
595 !!! warning
596
597 This can lead to security holes if another process obtains the same
598 file name and creates the file before you are able to. Open the file with
599 `JL_O_EXCL` if this is a concern. Using [`mktemp()`](@ref) is also
600 recommended instead.

801 """
802 walkdir(dir; topdown=true, follow_symlinks=false, onerror=throw)
803
804 Return an iterator that walks the directory tree of a directory.
805 The iterator returns a tuple containing `(rootpath, dirs, files)`.
806 The directory tree can be traversed top-down or bottom-up.
807 If `walkdir` encounters a [`SystemError`](@ref)
808 it will rethrow the error by default.
809 A custom error handling function can be provided through `onerror` keyword argument.
810 `onerror` is called with a `SystemError` as argument.
811
812 # Examples
813 ```julia
814 for (root, dirs, files) in walkdir(".")
815 println("Directories in \$root")
816 for dir in dirs
817 println(joinpath(root, dir)) # path to directories
818 end
819 println("Files in \$root")
820 for file in files
821 println(joinpath(root, file)) # path to files
822 end
823 end
824 ```
825
826 ```julia-repl
827 julia> mkpath("my/test/dir");
828
829 julia> itr = walkdir("my");
830
831 julia> (root, dirs, files) = first(itr)
832 ("my", ["test"], String[])
833
834 julia> (root, dirs, files) = first(itr)
835 ("my/test", ["dir"], String[])
836
837 julia> (root, dirs, files) = first(itr)
838 ("my/test/dir", String[], String[])
839 ```
840 """
841 function walkdir(root; topdown=true, follow_symlinks=false, onerror=throw)
842 content = nothing
843 try
844 content = readdir(root)
845 catch err
846 isa(err, SystemError) || throw(err)
847 onerror(err)
848 # Need to return an empty closed channel to skip the current root folder
849 chnl = Channel(0)
850 close(chnl)
851 return chnl
852 end
853 dirs = Vector{eltype(content)}()
854 files = Vector{eltype(content)}()
855 for name in content
856 path = joinpath(root, name)
857
858 # If we're not following symlinks, then treat all symlinks as files
859 if (!follow_symlinks && islink(path)) || !isdir(path)
860 push!(files, name)
861 else
862 push!(dirs, name)
863 end
864 end
865
866 function _it(chnl)
867 if topdown
868 put!(chnl, (root, dirs, files))
869 end
870 for dir in dirs
871 path = joinpath(root,dir)
872 if follow_symlinks || !islink(path)
873 for (root_l, dirs_l, files_l) in walkdir(path, topdown=topdown, follow_symlinks=follow_symlinks, onerror=onerror)
874 put!(chnl, (root_l, dirs_l, files_l))
875 end
876 end
877 end
878 if !topdown
879 put!(chnl, (root, dirs, files))
880 end
881 end
882
883 return Channel(_it)
884 end
